# Simulate Gegenbauer long-memory series

gegenbauer_sim <- function(n = 1500, u = 0.7, d = 0.35, sigma = 1, M = 50) {
   
  # White noise
  e <- rnorm(n + M, sd = sigma)
  
  # Compute Gegenbauer coefficients recursively
  psi <- numeric(M)
  psi[1] <- 1
  psi[2] <- 2*u*d
  
  for (k in 3:M) {
    psi[k] <- ((2*u*(k-2+d)*psi[k-1] -
                (k-3+2*d)*psi[k-2]) / (k-1))
  }
  
  # Apply filter (convolution)
  y <- filter(e, filter = psi, method = "convolution", sides = 1)
  
  # Remove burn-in
  y <- y[(M+1):(M+n)]
  return(as.numeric(y))
}

# k=2 GARMA model simulation best


gegenbauer_sim_k2 <- function(n = 1500,
                             u = c(0.7, 0.3),
                             d = c(0.35, 0.25),
                             sigma = 1,
                             M = 100) {
  
  # Function to compute psi coefficients for one factor
  compute_psi <- function(u, d, M) {
    psi <- numeric(M)
    psi[1] <- 1
    psi[2] <- 2 * u * d
    
    for (k in 3:M) {
      psi[k] <- (2 * u * (k - 2 + d) * psi[k - 1] -
                 (k - 3 + 2 * d) * psi[k - 2]) / (k - 1)
    }
    return(psi)
  }
  
  # Compute psi for both factors
  psi1 <- compute_psi(u[1], d[1], M)
  psi2 <- compute_psi(u[2], d[2], M)
  
  # Convolution of filters
  psi_total <- convolve(psi1, rev(psi2), type = "open")
  
  # Truncate to M (important for stability)
  psi_total <- psi_total[1:M]
  
  # White noise
  e <- rnorm(n + M, sd = sigma)
  
  # Apply filter
  y <- filter(e, filter = psi_total, method = "convolution", sides = 1)
  
  # Remove burn-in
  y <- y[(M + 1):(M + n)]
  
  return(as.numeric(y))
}

gegenbauer_2factor_sim <- function(n = 1500,
                                   u = c(0.3, 0.7),   # frequencies
                                   d = c(0.2, 0.3),   # memory parameters
                                   sigma = 1) {
  
  # Frequencies
  j <- 0:(n - 1)
  lambda <- 2 * pi * j / n
  
  # Initialize spectral density
  f_lambda <- rep(1, n)
  
  # Multiply Gegenbauer factors
  for (k in 1:2) {
    f_lambda <- f_lambda * abs(2 * (cos(lambda) - cos(2 * pi * u[k])))^(-2 * d[k])
  }
  
  # Scale
  f_lambda <- sigma^2 * f_lambda
  
  # Generate complex Gaussian noise
  Z_real <- rnorm(n)
  Z_imag <- rnorm(n)
  Z <- complex(real = Z_real, imaginary = Z_imag)
  
  # Apply spectral shaping
  X_freq <- sqrt(f_lambda) * Z
  
  # Inverse FFT to get time series
  x <- Re(fft(X_freq, inverse = TRUE)) / sqrt(n)
  
  return(x)
}

gegenbauer_sim_spec <- function(n = 1500, u = 0.7, d = 0.35, sigma = 1, taper = TRUE) {
  
  # Fourier frequencies (0 to 2π)
  j <- 0:(n-1)
  lambda <- 2*pi*j/n
  
  # Correct Gegenbauer spectral density
  f <- (1 - 2*u*cos(lambda) + u^2)^(-2*d)
  
  # Avoid numerical explosion at the pole
  f[!is.finite(f)] <- max(f[is.finite(f)])
  
  # Optional taper to reduce leakage
  if (taper) {
    taper_window <- 0.5 * (1 - cos(2*pi*(1:n)/n))  # cosine bell
  } else {
    taper_window <- rep(1, n)
  }
  
  # Generate complex Gaussian noise (Hermitian symmetry enforced)
  z_real <- rnorm(n)
  z_imag <- rnorm(n)
  z <- complex(real = z_real, imaginary = z_imag)
  
  # Impose spectrum
  y_fft <- sqrt(f) * z
  
  # Inverse FFT
  y <- Re(fft(y_fft, inverse = TRUE)) / sqrt(n)
  
  # Apply taper in time domain (optional)
  y <- y * taper_window
  
  # Standardise variance
  y <- sigma * scale(y)
  
  return(as.numeric(y))
}

# ###########################
# Version 1: FFT - based ( Spectral Domain )
# - Fast
# - Works naturally for freq = 0 (1st - order filter )
# - Peak at zero frequency only
# - Approximate spectral shape for non - zero frequency peaks
# ###########################

gegenbauer_sim_fft <- function (n, u, d, sigma) 
{
# 1. Fourier frequencies
j <- 0:(n -1)
lambda <- 2*pi*j/n

# 2. Spectral density (1st - order approximation )
f <- (1 - 2*u*cos( lambda ) + u^2)^(-2*d)

# 3. Random complex numbers
z <- complex (real = rnorm (n), imaginary = rnorm (n))

# 4. Impose spectrum
y_fft <- sqrt(f)*z

# 5. Inverse FFT to time domain
y <- Re(fft(y_fft, inverse = TRUE)) / sqrt(n)

# 6. Scale
y <- sigma*as.numeric(scale(y))
return (y)
}

# Version 2: Time - domain filter
# - Computes Gegenbauer coefficients recursively
# - Convolves with white noise
# - Can simulate true non - zero frequency peak
# - More accurate for cyclical long - memory
# ###########################

gegenbauer_sim_filter <- function(n, u, d, sigma, M =50) 
{
# 1. White noise with burn-in
e <- rnorm(n + M, sd = sigma)

# 2. Recursive coefficients
psi <- numeric(M)
psi[1] <- 1
psi[2] <- 2*u*d
if (M > 2) {
for (k in 3:M) {
psi[k] <- (2*u*(k-2+d)*psi[k-1] - (k-3+2*d)*psi[k-2])/(k-1)
               }
           } 

# 3. Filter via convolution
y <- stats::filter(e, filter=psi, method="convolution", sides =1)

# 4. Remove burn -in and scale
y <- y[(M+1):(M+n)]
y <- as.numeric(scale(y))
return(y)
}

# Version 4: Direct Spectral ( True Gegenbauer )
# - Uses f( lambda ) = |2( cos( lambda ) - u)|^{ -2d}
# - Accurate non - zero frequency peak
# - Fast using FFT
# ###########################
gegenbauer _sim_v4 <- function (n, u, d, sigma =1, epsilon =1e -8) {
# 1. Fourier frequencies
j <- 0:(n -1)
lambda <- 2*pi*j/n
# 2. True Gegenbauer spectral density
f <- (abs (2 * (cos( lambda ) - u)) + epsilon )^( -2*d) # add small
epsilon to avoid division by zero
# 3. White noise in complex plane
z <- complex ( real = rnorm (n), imaginary = rnorm (n))
# 4. Apply filter in frequency domain
y_fft <- sqrt (f) * z
# 5. Inverse FFT to time domain
y <- Re(fft(y_fft , inverse = TRUE )) / sqrt (n)
# 6. Scale
y <- sigma * as. numeric ( scale (y))
return (y)
} y
